create table venta (
idVenta	int IDENTITY(1,1),
codigoVenta varchar(45) PRIMARY KEY,
precioVenta float,
fechaVenta varchar(45),
horaVenta varchar(45),
tipoVenta varchar(45),
grupoVenta_Producto varchar(45),
codigoDistribuidores varchar(45),
codigoClientes varchar(45)
);


create table distribuidores (
id	int IDENTITY(1,1) PRIMARY KEY,
codigo varchar(45),
nombre varchar(45),
email varchar(45),
direccion varchar(45),
password varchar(45),
estado tinyint
);
